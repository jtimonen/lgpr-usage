2. Fitting a model¶

In [3]:
fit <- lgp(formula  = y ~ id + age + z1 + z2,
          data     = simData$data,
          iter     = 1000,
          control  = list(adapt_delta = 0.9),
          refresh  = 0,
          likelihood = "binomial",
          N_trials = rep(10, times = 80),
          verbose  = TRUE,
          parallel = TRUE)
